##yield

    def fab(max):
        n, a, b = 0, 0, 1
        while n < max:
            yield b
            a, b = b, a + b
            n += 1

    g = fab(30)

    for i in g:
        print i

### class method

    class A(object):
        def __init__(self, name):
            self.name = name

        def __str__(self):
            return "<%s>: %s" % (self.__class__, self.name)

        @classmethod
        def create(cls, name):
            return cls(name)


    class B(A):
        pass
    a = A.create('haha')

    def wrapper2(func):
        def f(args):
            print '##before %s:' % func.__name__

            print '##after %s:' % func.__name__
            return result
        return f

###with arguments

    def wrapper3(*arguments):
        def _wrapper3(function):
            def _fun_wrapper(*args, **kwargs):
